gisaxs_toolkit/               <-- 主项目文件夹

├── core/                    <-- 核心功能模块

│   ├── ml_model.py          <-- ML 模型定义、训练和预测
│   ├── dataset_builder.py   <-- 构建 trainset 的逻辑

├── model/                   <-- 存放机器学习模型文件（如 .h5, .pt, .pkl）
│
├── config/                  <-- 存储配置文件、超参数等

│
├── logs/                    <-- 日志输出目录（自动生成）
│
├── requirements.txt         <-- Python 依赖包列表
├── README.md                <-- 项目介绍
└── setup.py                 <-- 如果你计划将它打包成一个库或应用（可选）
